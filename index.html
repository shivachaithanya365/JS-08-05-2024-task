        }

        2x3=6
        2x4=8
        2x5=10

        2x7=14

2.print even numbers 0 to 10

------->for (a = 0; a <= 10; a++) {
            if (a % 2 == 0) {
                console.log(a);

            }
        }
        
------->output:

        0,2,4,6,8,10

        
3.print reverse of even numbers

------->for (a = 10; a >= 0; a--) {
            if (a % 2 == 0) {
                console.log(a);

            }
        }

------->output:

        10,8,6,4,2,0
